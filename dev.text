        <?php
function angkaKeHuruf($angka) {
    // Menangani angka di bawah 27 (1-26 -> 'a' - 'z')
    if ($angka >= 1 && $angka <= 26) {
        return chr(96 + $angka); // chr(97) = 'a', chr(98) = 'b', ...
    }

    // Menangani angka di atas 26 (27 -> 'AA', 28 -> 'AB', ...)
    $huruf = '';
    while ($angka > 0) {
        $angka--; // Mengurangi 1 agar mulai dari 0 (untuk konversi yang lebih tepat)
        $huruf = chr($angka % 26 + 65) . $huruf; // chr(65) = 'A', chr(66) = 'B', ...
        $angka = intdiv($angka, 26); // Pembagian dengan 26 untuk mencapai huruf kedua
    }

    return $huruf;
}

// Contoh penggunaan
echo angkaKeHuruf(1);   // Output: A
echo angkaKeHuruf(26);  // Output: Z
echo angkaKeHuruf(27);  // Output: AA
echo angkaKeHuruf(52);  // Output: AZ
echo angkaKeHuruf(53);  // Output: BA
echo angkaKeHuruf(702); // Output: ZZ
echo angkaKeHuruf(703); // Output: AAA
?>
